
Para activar el uso de Modules en html (type="module"):

<script type="module" defer src="main.js"></script>


Tipos de Export:

//Default se utiliza cuando solo se va a exportar 1 elemento

export default User;


//Cuando vas a exportar más de 1 elemente se saca el default.

export { printName, printAge }

//SE EXPORTA ESTO:

class User {
  constructor(name, age) {
    this.name = name
    this.age = age
  }
}

function printName(user) {
  console.log(`User name is ${user.age}`)
}

function printAge(user) {
  console.log(`User age is ${user.age}`)
} 


Import:

import User from '/user.js';
import { printName, printAge } from '/user.js';
